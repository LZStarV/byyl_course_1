real ans;
 
int fact(int x) {
    int n;
    if (x == 0) return 1;
    else return fact(x - 1) * x;
} 
 
int main(void) {
    int n;     
    int t; 
    int x; 
    int a[10]; 
    t = n = 10; 
    x = (x - 1) * x % 10 ^ x / (x + 1); 
    while (t >= 0) {
        a[t] = fact(a[t]);
        t = t - 1;
    }
    while (t <= n) {
        ans = ans * a[t] * 1;
        t = t + 1;
    };
    if (ans < 0) { 
        ans = 0; 
    } else {
        if (ans > 8) ans = 8;
    }
    if (ans != 8) ans = ans + 1;
    return 0;
}
